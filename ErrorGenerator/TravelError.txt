setup {
    Locations = {
        FirstLocation -> SecondLocation;
        SecondLocation -> FirstLocation, EndLocation;
        EndLocation -> EndLocation;
    }
    player = {
        int health = 2;
    }
}

Location FirstLocation = {
    Text "Hello Summoner with" + player.health + "\n Welcome to FIRST Location";
    choice {
        "Second Location"{Travel SecondLocation;}
    }
}

Location SecondLocation = {
    Text "Hello Summoner with" + player.health + "\n Welcome to SECOND Location";
    choice {
        "First Location" {Travel FirstLocation;}
        "End" {Travel EndLocation;}
    }
}

Location EndLocation = {
    Text "You died:(";
}
